set(TBAASAN_LIT_SOURCE_DIR ${CMAKE_CURRENT_SOURCE_DIR})

set(TBAASAN_TESTSUITES)

set(TBAASAN_TEST_ARCH ${TBAASAN_SUPPORTED_ARCH})
if(APPLE)
  darwin_filter_host_archs(TBAASAN_SUPPORTED_ARCH TBAASAN_TEST_ARCH)
endif()

foreach(arch ${TBAASAN_TEST_ARCH})
  set(TBAASAN_TEST_TARGET_ARCH ${arch})
  string(TOLOWER "-${arch}" TBAASAN_TEST_CONFIG_SUFFIX)
  get_test_cc_for_arch(${arch} TBAASAN_TEST_TARGET_CC TBAASAN_TEST_TARGET_CFLAGS)
  string(TOUPPER ${arch} ARCH_UPPER_CASE)
  set(CONFIG_NAME ${ARCH_UPPER_CASE}Config)

  configure_lit_site_cfg(
    ${CMAKE_CURRENT_SOURCE_DIR}/lit.site.cfg.in
    ${CMAKE_CURRENT_BINARY_DIR}/${CONFIG_NAME}/lit.site.cfg)
  list(APPEND TBAASAN_TESTSUITES ${CMAKE_CURRENT_BINARY_DIR}/${CONFIG_NAME})
endforeach()

set(TBAASAN_TEST_DEPS ${SANITIZER_COMMON_LIT_TEST_DEPS})
if(NOT COMPILER_RT_STANDALONE_BUILD)
  list(APPEND TBAASAN_TEST_DEPS tbaasan)
endif()

add_lit_testsuite(check-tbaasan "Running the TBAASanitizer tests"
  ${TBAASAN_TESTSUITES}
  DEPENDS ${TBAASAN_TEST_DEPS}
  )
set_target_properties(check-tbaasan PROPERTIES FOLDER "Compiler-RT Misc")
